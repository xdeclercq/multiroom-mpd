Resource 'zone'

A zone.

Properties

  • name: the zone name

Example

{
  "_links": {
    "self": {
      "href": "http://localhost:8080/multiroom-mpd/api/zones/Kitchen"
    },
    "mr:player": {
      "href": "http://localhost:8080/multiroom-mpd/api/zones/Kitchen/player"
    }
  },

  "name": "Kitchen"
}
